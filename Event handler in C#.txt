using System;

public class HelloWorld
{
    public static void Main(string[] args)
    {
        // Create a button
        Button myButton = new Button();

        // Subscribe to the Click event using +=
        myButton.Click += MyEventHandler;

        // Simulate clicking the button
        myButton.OnClick();

        Console.ReadLine();
    }

    // Event handler method (follows the standard pattern)
    static void MyEventHandler(object sender, EventArgs e)
    {
        Console.WriteLine("Button was clicked!");
    }
}

// Simple Button class with an event
class Button
{
    // Declare an event using EventHandler delegate
    public event EventHandler Click;

    // Method to raise (trigger) the event
    public void OnClick()
    {
        // Raise event (if anyone subscribed)
        if (Click != null)
        {
            Click(this, EventArgs.Empty);
        }
    }
}
